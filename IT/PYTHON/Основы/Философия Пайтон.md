Python императивный или декларативный язык?  :: Пайтон мультипарадигменный язык, лучше императивный подход применять к сложным бизнес-процессам, соблюдая философия пайтона "явное лучше, чем неявное", "Простое лучше сложного", а вот к простым конструкциям лучше применять декларативный подход. императивный отвечает на вопрос как, а декларативный на вопрос что? - ориентация на результат

```python
# Идеальный Python код сочетает оба подхода
class DataProcessor:
    def __init__(self, data):
        self.data = data
    
    # Императивно для сложной логики
    def process_complex_logic(self):
        results = []
        for item in self.data:
            # Сложные условия и преобразования
            if self._is_valid(item) and self._should_process(item):
                processed = self._transform_item(item)
                results.append(processed)
        return results
    
    # Декларативно для простых преобразований
    def get_filtered_data(self):
        return [
            self._transform_item(item)
            for item in self.data
            if self._is_valid(item)
        ]
    
    def _is_valid(self, item):
        return item is not None and isinstance(item, (int, float))
    
    def _should_process(self, item):
        return item > 0
    
    def _transform_item(self, item):
        return item ** 2

# Использование
processor = DataProcessor([1, 2, None, 3, -1, 4.5])
print("Complex processing:", processor.process_complex_logic())
print("Simple filtering:", processor.get_filtered_data())
```
