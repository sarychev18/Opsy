```dataviewjs
const categoryColors = {
    "эффективность": "#fff9f2", "продажи": "#ebffff", "default": "#f5f5f5"
};
const getCategoryColor = category => {
    if (category in categoryColors) return categoryColors[category];
    else return categoryColors["default"];
};

const getBooks = () => dv.pages("").where(page => {
    if (page["Тип"] && page["Тип"].path == "types/Книга.md" && page.file.folder != "templates" && !page["Закончил читать"] && page["Начал читать"]) {
      return page;
    }
}).sort(p => p["Начал читать"], "desc");

// ФИКС: Безопасное форматирование даты
const formatDate = (dateValue) => {
    if (!dateValue) return "не указано";
    
    try {
        const date = dv.date(dateValue);  // Преобразуем в Luxon
        if (!date || !date.isValid) return "некорректная дата";
        
        const now = dv.date("now");
        return date.year === now.year
            ? date.toFormat("d.MMMM", { locale: "ru" })
            : date.toFormat("d.MMMM.yyyy", { locale: "ru" });
    } catch (error) {
        return String(dateValue).substring(0, 10);
    }
};

let books = [];

for (const book of getBooks()) {
    // ФИКС: Защита от null для Категории
    const categoriesArray = book.Категории ? (Array.isArray(book.Категории) ? book.Категории : []) : [];
    const categories = categoriesArray.map(
        c => `<span class="category" style="background-color: ${getCategoryColor(c)}">${c}</span>`
    ).join(" ");
    
    // ФИКС: Безопасные значения
    const coverUrl = book.Обложка || "";
    const progress = book.Progress ? `${book.Progress}` : "";
    const pageCount = book.Страниц ? `${book.Страниц} стр.` : "0 стр.";
    
    books.push(`<div class="book">
        <a data-tooltip-position="top" data-href="${book.file.name}" href="${book.file.name}.md" class="internal-link" target="_blank" rel="noopener nofollow">
            <img src="${coverUrl}" data-filename="${book.file.name}" 
                 ${coverUrl ? '' : 'style="display: none;"'}
                 onerror="this.style.display='none';" />
        </a>
        ${progress}
        начал ${formatDate(book["Начал читать"])}<br>
        <div class="categories">${categories}</div>
        <div class="pages">${pageCount}</div>
    </div>`);
}

dv.el("div", books.join(""), {cls: "books"});
```

## Все книги
```dataviewjs
let booksCount = 0;


let bookPages = dv.pages("").where(page => {
    if (page["Тип"] && page["Тип"].path == "types/Книга.md" && page.file.folder != "Шаблоны" && page["Начал читать"]) {
        booksCount++;
        return page;
    }
}).sort(p => p["Начал читать"], "desc");

const MAX_BOOK_NAME_LENGTH = 50;
let index = booksCount + 1;

dv.table(
    ["", "Название", "Автор", "Страниц", "Начал", "Прогресс"],
    bookPages.map(p => {
        const shortName = p.Название.length > MAX_BOOK_NAME_LENGTH  
            ? p.Название.substring(0, MAX_BOOK_NAME_LENGTH) + "..." 
            : p.Название;
        const authors = p.Автор.split(",").map(el => el.trim()).join("<br />");
        return [
            --index,
            dv.fileLink(p.file.path, false, shortName),
            authors,
            p.Страниц,
            p["Начал читать"] !== null ? p["Начал читать"] :
            "",
            p.Progress
        ];
    })
);
```
